  function zatwierdzFormularz(nazwaFormularza, zdarzenie) {
    var formularz = document.forms[nazwaFormularza];

    if (formularz.action.indexOf("event")==-1) {
      formularz.action = akcja + "?event=" + zdarzenie + "&href=#hrefKryteriumFiltrowania";
    } else {
      formularz.action = akcja + "&event=" + zdarzenie + "&href=#hrefKryteriumFiltrowania";
